String Status = "mainmodStatus";

#include <Wire.h>
#include "ASOLED.h"


void CLS() {
LD.init();
LD.clearDisplay();
}

void mainmod() {
LD.printString_6x8("Hello, World!", 0, 0);
if (digitalRead(2) == 0) {
CLS();
Status = "byeStatus";
}
}

void bye() {
LD.printString_6x8("Goodbye, World!", 0, 0);
if (digitalRead(3) == 0) {
CLS();
Status = "mainmodStatus";
}
}


void setup() {
LD.init();
LD.clearDisplay();

mainmod();

pinMode(2, INPUT_PULLUP);
pinMode(3, INPUT_PULLUP);
}

void loop() {
if (Status == "mainmodStatus") {
mainmod();
}
if (Status == "byeStatus") {
bye();
}

}
